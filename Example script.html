 23     var m is integer;
 24     var n is integer;
 25     return integer;

 27   begin

 43   function multy(n)
 44     var n is integer;

 46   is

 48     var i is integer;

 50     var g is grid;
 51     var l is line;

 53     i := 0;

 55     while i < 10 do
 56       j := 0;
 57       // 无论何种条件，进入循环体
 58       repeat
 59         g[i][j] := i * j;

 61       until j >= 10;
 62       i := i + 1;
 63     // 此时回到循环的条件判断处
 64     end while
 65     // 遍历 g 中的每个元素
 66     foreach l in g do

 70     end foreach
